CREATE TABLE members ( id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, password VARCHAR(255) NOT NULL, email VARCHAR(255), phone VARCHAR(20), birthday DATE, talanton INT DEFAULT 0, join_date DATE, service VARCHAR(50), area VARCHAR(100), status VARCHAR(20) DEFAULT 'active', sequential_number INT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP ); CREATE TABLE products ( id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, price INT NOT NULL, stock INT DEFAULT 0, category VARCHAR(100), description TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP ); CREATE TABLE occasions ( id INT AUTO_INCREMENT PRIMARY KEY, title VARCHAR(255) NOT NULL, date DATE, time TIME, location VARCHAR(255), points INT DEFAULT 0, type VARCHAR(100), description TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP ); CREATE TABLE attendance ( id INT AUTO_INCREMENT PRIMARY KEY, member_id INT, member_name VARCHAR(255), date DATE, type VARCHAR(50), notes TEXT, timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (member_id) REFERENCES members(id) ON DELETE CASCADE ); CREATE TABLE talanton_history ( id INT AUTO_INCREMENT PRIMARY KEY, member_id INT, member_name VARCHAR(255), amount INT, type VARCHAR(20), reason VARCHAR(255), date DATE, timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (member_id) REFERENCES members(id) ON DELETE CASCADE ); CREATE TABLE settings ( id INT PRIMARY KEY DEFAULT 1, daily_attendance_points INT DEFAULT 1, birthday_alert_days INT DEFAULT 7, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP );
